Differences of Two Proportions

  • Variances of independent variables add

\[SE_{\hat{p}_1 - \hat{p}_2} = \sqrt{SE_{\hat{p}_1}^2 + SE_{\hat{p}_2}^2} = \sqrt{\dfrac{p_1 (1-p_1)}{n_1} + \dfrac{p_2 (1-p_2)}{n_2}}\]

  • Even if you’re subtracting two noisy measurements, the uncertainty from both still adds up
    • Think of it like using two shaky rulers: subtracting doesn’t cancel the shakiness — it just combines it
set.seed(1)       # ensures reproducibility
B <- 1000          # number of simulations
n <- 500           # sample size per group
p1 <- 0.5          # true proportion in group 1
p2 <- 0.4          # true proportion in group 2

# Create empty vectors to store simulated means and SDs
mean_x1 <- mean_x2 <- numeric(B)
sd_x1 <- sd_x2 <- numeric(B)

# Loop to simulate samples for both groups
for (i in 1:B) {
  
  # Generate random binary outcomes for group 1 (success = 1, failure = 0)
  x1 <- rbinom(n, size = 1, prob = p1)
  mean_x1[i] <- mean(x1)  # sample proportion for group 1
  sd_x1[i]   <- sd(x1)    # sample SD for group 1
  
  # Repeat for group 2
  x2 <- rbinom(n, size = 1, prob = p2)
  mean_x2[i] <- mean(x2)
  sd_x2[i]   <- sd(x2)
  
}

# Theoretical standard error for a sample proportion
sqrt(p1*(1-p1)/n)
[1] 0.02236068
# Empirical standard error from simulated data
mean(sd_x1) / sqrt(n)
[1] 0.0223613
# Compute 95% confidence bounds around the sampling mean
ub_x1 <- mean(mean_x1) + 1.96 * sqrt(p1*(1-p1)/n)
lb_x1 <- mean(mean_x1) - 1.96 * sqrt(p1*(1-p1)/n)

# Plot histogram of simulated sample proportions
par(mar = c(4.5, 4.5, 1, 1))
hist(mean_x1, breaks = 12, 
     main = "",
     xlab = "Sample Proportion (Group 1)",
     cex.axis = 1.25, cex.lab = 1.5)
abline(v = c(lb_x1, ub_x1), lty = 2, lwd = 2)  # add CI bounds
# Compute simulated differences
diff <- mean_x1 - mean_x2

# Theoretical SE for the difference in proportions
sqrt(p1*(1-p1)/n + p2*(1-p2)/n)
[1] 0.03130495
# Empirical SE estimate (not exact but illustrative)
mean(sd_x1 + sd_x2) / sqrt(n + n)
[1] 0.03129505
# Compute 95% CIs using correct (additive) and incorrect (subtractive) formulas
## correct
ub_diff_1 <- mean(diff) + 1.96 * sqrt(p1*(1-p1)/n + p2*(1-p2)/n)
lb_diff_1 <- mean(diff) - 1.96 * sqrt(p1*(1-p1)/n + p2*(1-p2)/n)

## incorrect
ub_diff_2 <- mean(diff) + 1.96 * sqrt(p1*(1-p1)/n - p2*(1-p2)/n)
lb_diff_2 <- mean(diff) - 1.96 * sqrt(p1*(1-p1)/n - p2*(1-p2)/n)

# Visualize sampling distribution of the difference
hist(diff, breaks = 12, 
     main = "",
     xlab = "Difference in Sample Proportions (p1 - p2)",
     cex.axis = 1.25, cex.lab = 1.5)

# Add both sets of CI bounds
abline(v = c(lb_diff_1, ub_diff_1), lty = 2, lwd = 2)
abline(v = c(lb_diff_2, ub_diff_2), lty = 3, lwd = 2, col = "red")

# legend
legend("topright",
       legend = c("SE1 + SE2 (correct)", "SE1 - SE2 (incorrect)"),
       lwd = 2, lty = c(2, 3),
       col = c("black", "red"),
       bty = "n")
# Check coverage rates for both formulas
mean(ifelse(diff >= ub_diff_1 | diff <= lb_diff_1, 1, 0))
[1] 0.037
mean(ifelse(diff >= ub_diff_2 | diff <= lb_diff_2, 1, 0))
[1] 0.793

Differences of Two Proportions: Example 1

  • Study of ICU patients admitted after a heart attack
  • Randomly assigned:
    • Treatment: blood thinner
    • Control: no blood thinner
  • Outcome measured: 24-hour survival
# read in data
x <- read.csv("data/cpr.csv")

# convert to matrix
z <- as.data.frame.matrix(table(x))
z$total <- rowSums(z)
z <- rbind(z, total = colSums(z))
z
          died survived total
control     39       11    50
treatment   26       14    40
total       65       25    90
## calculate n for each group
nt <- z$total[2]
nc <- z$total[1]

## calculate p(survive) for each group
pt <- z$survived[2] / nt
pc <- z$survived[1] / nc

## calculate test statistics
point_est <- pt - pc
se <- sqrt((pt * (1 - pt) / nt) + (pc * (1 - pc) / nc))

z <- data.frame(
  sig_level = c(0.01, 0.05, 0.1),
  z_score = c(2.576, 1.96, 1.645)
)

z$min <- point_est - z$z_score * se
z$max <- point_est + z$z_score * se
z
  sig_level z_score         min       max
1      0.01   2.576 -0.11599788 0.3759979
2      0.05   1.960 -0.05717230 0.3171723
3      0.10   1.645 -0.02709104 0.2870910
par(mar = c(4.5, 4.5, 1, 1))
plot(1:3, rep(point_est, 3), 
     ylim = c(min(z$min), max(z$max)),
     xlab = "Significance Level (α)",
     ylab = "Est. with Conf. Int.",
     cex.axis = 1.25,
     cex.lab = 1.5,
     xaxt = "n")
axis(1, at = 1:3, labels = c(z$sig_level), cex.axis = 1.25)
abline(h = 0, lty = 2, lwd = 2)
arrows(1:3, z$min, 1:3, z$max, length = 0.05, angle = 90, code = 3)

  • We are 90% confident that blood thinners change the 24-hour survival rate by between -3 and 29 percentage points for patients similar to those in the study

  • Because 0% is within this range, the evidence is inconclusive

    • We cannot determine whether blood thinners help or harm heart attack patients who have undergone CPR

Differences of Two Proportions: Example 2

    • 5-year clinical trial: Do fish oil supplements reduce heart attack risk?
  • Participants randomly assigned:
    • Fish oil
    • Placebo
  • Compare heart attack rates across groups
# read in data
x <- read.csv("data/fish_oil.csv")

# convert to matrix
z <- as.data.frame.matrix(table(x))
z$total <- rowSums(z)
z <- rbind(z, total = colSums(z))
z
         hospital no_event total
fish oil      145    12788 12933
placebo       200    12738 12938
total         345    25526 25871
## calculate n for each group
nt <- z$total[1]
nc <- z$total[2]

## calculate p(hospital) for each group
pt <- z$hospital[1] / nt
pc <- z$hospital[2] / nc

## calculate test statistics
point_est <- pt - pc
se <- sqrt((pt * (1 - pt) / nt) + (pc * (1 - pc) / nc))

z <- data.frame(
  sig_level = c(0.01, 0.05, 0.1),
  z_score = c(2.576, 1.96, 1.645)
)

z$min <- point_est - z$z_score * se
z$max <- point_est + z$z_score * se

# convert to percentage points
point_est <- point_est * 100
z[,3:4] <- z[,3:4] * 100

par(mar = c(4.5, 4.5, 1, 1))
plot(1:3, rep(point_est, 3), 
     ylim = c(min(z$min), 0),
     xlab = "Significance Level (α)",
     ylab = "Est. with Conf. Int. (Pct. Points)",
     cex.axis = 1.25,
     cex.lab = 1.5,
     xaxt = "n")
axis(1, at = 1:3, labels = c(z$sig_level), cex.axis = 1.25)
abline(h = 0, lty = 2, lwd = 2)
arrows(1:3, z$min, 1:3, z$max, length = 0.05, angle = 90, code = 3)
  • Estimated effect: –0.0043
    • Heart attacks were 0.43 percentage points lower with fish oil
  • 90% confidence interval: –0.66 to –0.19 percentage points
    • Because 0 is not in the interval, the reduction is statistically significant

Practical vs. Statistical Significance

  • The effect is statistically significant, but very small:

    • about 0.4 fewer heart attacks per 100 people.

  • In large samples, even tiny effects can be statistically significant.

  • Practical question: Is such a small reduction worth the costs, side effects, or effort of treatment?

# effect size relative to mean
## a 27.5% reduction!
point_est / pc
[1] -27.47197

Differences of Two Proportions: Example 3

  • A drone quadcopter company is considering a new manufacturer for rotor blades
    • The new manufacturer is more expensive but claims that their higher-quality blades are 3% more reliable
  • Set up the appropriate hypotheses for this test:
    • \(H_0: p_{\text{highQ}} - p_{\text{standard}} = 0.03\)
    • \(H_A: p_{\text{highQ}} - p_{\text{standard}} \neq 0.03\)
  • Using these data, evaluate the hypotheses above at a significance level of 5%.

Compute the Point Estimate and Standard Error

x <- read.csv("data/quad.csv")

p_us <- sum(ifelse(x$prospective_supplier == "pass", 1, 0)) / 1000
p_them <- sum(ifelse(x$current_supplier == "pass", 1, 0)) / 1000

point_est <- p_us - p_them
se <- sqrt( p_us * (1 - p_us) / 1000 +  p_them * (1 - p_them) / 1000 )

z <- data.frame(
  sig_level = c(0.01, 0.05, 0.10),
  z_score = c(2.576, 1.96, 1.645)
)

z$min <- (point_est - z$z_score * se)
z$max <- (point_est + z$z_score * se)

# plot estimates with conf. intervals
par(mar = c(4.5, 4.5, 1, 1))
plot(1:3, rep(point_est, 3), 
     ylim = c(min(z$min), max(z$max)),
     xlab = "Significance Level (α)",
     ylab = "Est. with Conf. Int.",
     cex.axis = 1.25,
     cex.lab = 1.5,
     xaxt = "n")
axis(1, at = 1:3, labels = c(z$sig_level), cex.axis = 1.25)
abline(h = 0.03, lty = 2, lwd = 2)
arrows(1:3, z$min, 1:3, z$max, length = 0.05, angle = 90, code = 3)
## z-score
z <- (point_est - 0.03) / se

# simulate null hypothesis
set.seed(1)
sim <- rnorm(10000, mean = 0.03, sd = se)

# Probability of observing a value this extreme or larger
1 - mean(ifelse(point_est >= sim, 1, 0))
[1] 0.0049
# p-value (right-tailed)
1 - pnorm(z)
[1] 0.005648044
# visualize the result
par(mar = c(4.5, 4.5, 1, 1))
hist(sim, main = "", xlab = "Simulated Estimates",
     cex.lab = 1.5, cex.axis = 1.25)
abline(v = point_est, lty = 2)
legend("topleft",
       legend = c("Point Est."),
       lty = 2, cex = 1.25, bty = "n")
  • From the standard normal distribution:
    • The right-tail area is approximately 0.005
    • Doubling for a two-tailed test gives p = 0.01
    • Since p = 0.01 < 0.05, we reject the null hypothesis
  • We find statistically significant evidence that the higher-quality blades have a pass rate greater than 3% higher than the standard blades — exceeding the company’s claims.
